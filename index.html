from cryptography.fernet import Fernet
import base64
import os
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.backends import default_backend

def gen_key(pword: str) -> bytes:
    password = pword.encode()
    salt = b'\x8d\x1e\x1c\x9b\x90\xf2\x1a\x1b\x9d\x0e\x1c\x1f\x0e\x1c\x1f\x0e'
    kdf = PBKDF2HMAC(algorithm=hashes.SHA256(), length=32, salt=salt, iterations=?????, backend=default_backend())
    key = base64.urlsafe_b64encode(kdf.derive(password))
    return key
def coder(message: str, key: bytes) -> str:
    f = Fernet(key)
    res = f.encrypt(message.encode())
    return res.decode()

def decoder(message: str, key: bytes) -> str:
    f = Fernet(key)
    res = f.decrypt(message.encode())
    return res.decode()

SECRET_KEY = ''
encryption_key = gen_key(SECRET_KEY)

encrypted_text = coder('text', encryption_key)
print(f'result: {encrypted_text}')

decrypted_text = decoder('', encryption_key)
print(f'result: {decrypted_text}')
